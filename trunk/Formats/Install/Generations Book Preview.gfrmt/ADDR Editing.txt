load LineLabel

if #msg="wiki"

  newline
  if #vpos$>#boxend
    DisplayWidth #oldWidth
  endif

  hskip 12
  cell Static,local(@tag)&": "
  sizetofit
  hskip 0
  #leftSkip=#hpos$

  cell StaticField
  set width -1 border no offset 2 minheight 16
  #addr=@contents
  if #addr==""
    #addr="The address is not documented"
  endif
  text #addr
  sizetofit

  hideexcess
  #leftSkip=#leftSkip$

else

  gosub LineLabel local(@tag)
  offset #editOffset

  ! edit address in multiline field
  cell EditField
  set Width -1 minheight 44 maxheight 128
  sizetofit

  hideexcess
endif
