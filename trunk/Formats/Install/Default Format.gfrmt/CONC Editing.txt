  else if #msgnum>0
    ! use separate pane when not printing
    cell EditPane,"("&local("type to enter notes")&")"
    #fraction=#msg
    if #fraction<.5
      #fraction=.5
    endif

      position right
    else if #posNotes >1.5
      position top
    else if #posNotes >0.5
      position left
    else

    endif

    help local("Enter free-format notes")

     newline
     cell static,"Extra Text (should be copied to main notes text)"
     sizetofit
     newline
     cell EditField
     set maxheight 150 width -1
     sizetofit
  endif
