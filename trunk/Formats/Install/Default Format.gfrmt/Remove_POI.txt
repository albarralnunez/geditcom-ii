tell application "GEDitCOM II"    set message visible to false	set sm to scriptMessage	set boxnum to word 1 of sm	set poinum to word 2 of sm	tell front document		begin undo		set recRef to key record		tell recRef			set pois to find structures tag "_BOX" output "references"			set theBox to item boxnum of pois			tell theBox				set pois to find structures tag "_POI"				set pnum to (count of pois)				if pnum = 0 then					-- zero phones means nothing to remove				else					-- add one after current one					set snum to item 2 of item poinum of pois					delete structure snum				end if			end tell		end tell		end undo action local string for key "Remove Point of Interest"	end tellend tell
