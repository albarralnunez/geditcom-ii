tell application "GEDitCOM II"    set message visible to false
    set newText to local string for key "New Address"	tell front document		begin undo		set recRef to key record		tell recRef			set resiRef to (make new structure with properties {name:"RESI"})			tell resiRef				make new structure with properties {name:"ADDR", contents:newText}			end tell		end tell		end undo action local string for key "Add New Residence"	end tellend tell
