property scriptName : "Name Case Script"tell application "GEDitCOM II"	set message visible to false	tell front document		set details to editing details				-- exit if empty structure		if number of items in details < 3 then			beep			set msg to "The name is empty and therefore case cannot be changed"			display dialog msg buttons {"OK"} default button "OK" with title scriptName			return		end if				-- change name case		set nameStruct to item 3 of details		tell nameStruct			set oldName to contents			set newName to format name value oldName case "title"			set newName to format name value newName case "uppersurname"			set contents to newName		end tell	end tell
end tell
