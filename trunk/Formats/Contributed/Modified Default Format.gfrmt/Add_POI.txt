tell application "GEDitCOM II"    set message visible to false	set sm to scriptMessage	set boxNum to word 1 of sm	set poiNum to word 2 of sm	tell front document		begin undo		set recRef to key record
		tell recRef			set maps to find structures tag "_BOX" output "references"
			set theBox to item boxNum of maps			tell theBox				set pois to find structures tag "_POI"				set pnum to (count of pois)								set newPOI to {name:"_POI", contents:"Point of Interest"}				if pnum = 0 then					-- zero pois means none where there before					make new structure with properties newPOI				else					-- add one after current one					set snum to item 2 of item poiNum of pois					make new structure at after structure snum with properties newPOI				end if			end tell		end tell		end undo action local string for key "Add Point of Interest"	end tellend tell
