tell application "GEDitCOM II"	set message visible to false	tell front document		-- First gather some information about the selected record		set props to properties		set keyRec to (key record of props)	end tell				tell keyRec		set curdisp to {}		try			set curdisp to structure named "_DISP"		end try		if curdisp is not {} then			delete curdisp		end if	end tell	end tell
