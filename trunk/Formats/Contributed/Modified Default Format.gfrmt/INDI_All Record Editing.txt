!SJGR changed from default to adjust tabs for extra columns events table

load TagBlock,ResetTabs,ListBlock,References,AttachData,INDIPanes

sub TableTabs,#check
  newline
  ClearTabs
  #lineSkip=2
  #cellSkip=2
  #tableWidth=#rightMargin$-#rightSkip-#hpos$-#cellSkip-48  !changed from 21 to account for extra space needed for 2 icons SJGR
  #nameWidth=(#tableWidth-2*#cellSkip-20)/2
  if #nameWidth>180
    #nameWidth=180
  endif
  if #check=1
    !Tabs Changed to account for place detail in main window
    !SetTab #habs$+#nameWidth,#habs$+#nameWidth+20,#habs$+#tableWidth
    SetTab #habs$+#nameWidth,#habs$+#nameWidth+20,#habs$+#nameWidth+222,#habs$+#tableWidth
  else if #check=0
    SetTab #habs$+#nameWidth+20,(#habs$+#tableWidth)/1.7,#habs$+#tableWidth
   else if #check=3 !for RESI tags
    SetTab #habs$+#nameWidth+20,#habs$+#tableWidth
  else
    SetTab #habs$+#nameWidth+20,#habs$+#tableWidth-22,#habs$+#tableWidth
  endif
  #first="yes"
endsub
